# Skill: API Design

## Purpose
This skill defines REST API conventions for all Express routes generated by
the backend agent. Every route must conform to these patterns.

---

## Response envelope

All responses use this shape — no exceptions:

```typescript
// Success
{ "success": true, "data": <payload> }

// Success with pagination
{
  "success": true,
  "data": [...],
  "meta": {
    "total": 142,
    "page": 1,
    "perPage": 20,
    "hasNextPage": true,
    "nextCursor": "clp_xyz..."
  }
}

// Error
{
  "success": false,
  "error": {
    "message": "Human-readable description",
    "code": "MACHINE_READABLE_CODE",
    "details": [{ "field": "email", "message": "Must be a valid email" }]
  }
}
```

---

## Status code conventions

| Situation | Status |
|-----------|--------|
| GET/PUT success | 200 |
| POST created | 201 |
| DELETE success | 200 (with `{ success: true, data: { id } }`) |
| Validation error | 400 |
| Unauthenticated | 401 |
| Forbidden | 403 |
| Not found | 404 |
| Conflict (duplicate) | 409 |
| Rate limited | 429 |
| Server error | 500 |

Never return 200 for errors. Never return 500 for client errors.

## Route structure

```typescript
// routes/task.routes.ts
import { Router } from 'express'
import { authenticate } from '../middleware/auth'
import { validate } from '../middleware/validate'
import { createTaskSchema, updateTaskSchema } from '../schemas/task.schema'
import * as TaskController from '../controllers/task.controller'

const router = Router()

router.use(authenticate) // apply auth to all task routes

router.get('/',            TaskController.list)
router.post('/', validate(createTaskSchema), TaskController.create)
router.get('/:id',         TaskController.get)
router.patch('/:id', validate(updateTaskSchema), TaskController.update)
router.delete('/:id',      TaskController.remove)
router.post('/:id/assign', TaskController.assign)

export default router
```

## Controller pattern

```typescript
// controllers/task.controller.ts
import { asyncHandler } from '../utils/asyncHandler'
import * as TaskService from '../services/task.service'
import { AppError } from '../utils/errors'

export const create = asyncHandler(async (req, res) => {
  const task = await TaskService.create({
    ...req.body,
    creatorId: req.user.id,
  })
  res.status(201).json({ success: true, data: task })
})

export const get = asyncHandler(async (req, res) => {
  const task = await TaskService.findById(req.params.id, req.user.id)
  if (!task) throw new AppError('Task not found', 404, 'NOT_FOUND')
  res.json({ success: true, data: task })
})

export const remove = asyncHandler(async (req, res) => {
  const { id } = await TaskService.softDelete(req.params.id, req.user.id)
  res.json({ success: true, data: { id } })
})
```

## Service pattern

```typescript
// services/task.service.ts
import { prisma } from '../prisma/client'
import { logActivity } from './activity.service'
import { AppError } from '../utils/errors'

export async function create(data: CreateTaskData) {
  const task = await prisma.task.create({
    data,
    include: { assignee: true, labels: true, column: true }
  })

  await logActivity({
    type: 'TASK_CREATED',
    taskId: task.id,
    workspaceId: task.workspaceId,
    userId: data.creatorId,
    meta: { title: task.title }
  })

  return task
}

export async function softDelete(id: string, userId: string) {
  const task = await prisma.task.findUnique({ where: { id } })
  if (!task) throw new AppError('Task not found', 404)
  if (task.creatorId !== userId) throw new AppError('Forbidden', 403)

  return prisma.task.update({
    where: { id },
    data: { deletedAt: new Date() },
    select: { id: true }
  })
}
```

## Zod schema pattern

```typescript
// schemas/task.schema.ts
import { z } from 'zod'

export const createTaskSchema = z.object({
  title: z.string().min(1, 'Title is required').max(255),
  description: z.string().max(10_000).optional(),
  columnId: z.string().cuid('Invalid column ID'),
  priority: z.enum(['URGENT', 'HIGH', 'MEDIUM', 'LOW', 'NONE']).default('NONE'),
  dueDate: z.string().datetime().optional(),
  assigneeId: z.string().cuid().optional(),
  labelIds: z.array(z.string().cuid()).default([]),
})

export const updateTaskSchema = createTaskSchema.partial()

export type CreateTaskData = z.infer<typeof createTaskSchema>
export type UpdateTaskData = z.infer<typeof updateTaskSchema>
```

## Pagination pattern (cursor-based)

```typescript
// Always use cursor-based pagination for lists
const tasks = await prisma.task.findMany({
  where: { columnId, deletedAt: null },
  take: perPage + 1, // fetch one extra to check hasNextPage
  cursor: cursor ? { id: cursor } : undefined,
  orderBy: { createdAt: 'desc' },
  include: { assignee: { select: { id: true, name: true, avatar: true } } }
})

const hasNextPage = tasks.length > perPage
const data = hasNextPage ? tasks.slice(0, -1) : tasks
const nextCursor = hasNextPage ? data[data.length - 1].id : null

res.json({
  success: true,
  data,
  meta: { hasNextPage, nextCursor, perPage }
})
```

## Security conventions

- Validate ownership before every mutate operation (user must belong to workspace)
- Never return password hash in any response — use `select` to exclude it
- Never use `findFirst` without a `where` clause that scopes to the user
- Always parameterise queries — Prisma handles this, never concatenate strings
- Log but don't expose internal error details: message only, no stack traces
