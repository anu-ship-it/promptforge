# Skill: Test Writing

## Purpose
This skill defines testing conventions for both frontend (Vitest) and backend
(Jest) test suites generated by the tester agent.

---

## Core principles

1. **Test behaviour, not implementation** ã don't test that a function was called,
   test that the correct output appeared or the correct change happened.
2. **One assertion per test where possible** ã if a test fails you know exactly why.
3. **Arrange ã Act ã Assert** ã every test follows this structure.
4. **Tests are documentation** ã someone reading only the test suite should
   understand what the system does.

## Backend test conventions (Jest + Supertest)

### Test file structure

```typescript
import request from 'supertest'
import { app } from '../output/backend/src/app'
import { prisma, createTestUser, createTestWorkspace } from './setup'

describe('Task API', () => {
  // Helper: create authenticated user and return token
  let token: string
  let userId: string
  let workspaceId: string

  beforeEach(async () => {
    const user = await createTestUser()
    userId = user.id
    token = user.token
    const workspace = await createTestWorkspace(userId)
    workspaceId = workspace.id
  })

  describe('POST /api/v1/tasks', () => {
    it('creates a task and returns 201 with the task data', async () => {
      const res = await request(app)
        .post('/api/v1/tasks')
        .set('Authorization', `Bearer ${token}`)
        .send({ title: 'Fix the login bug', columnId: 'col_123', priority: 'HIGH' })

      expect(res.status).toBe(201)
      expect(res.body.success).toBe(true)
      expect(res.body.data).toMatchObject({
        title: 'Fix the login bug',
        priority: 'HIGH',
        creatorId: userId
      })
      expect(res.body.data.id).toBeDefined()
    })

    it('returns 400 when title is missing', async () => {
      const res = await request(app)
        .post('/api/v1/tasks')
        .set('Authorization', `Bearer ${token}`)
        .send({ columnId: 'col_123' })

      expect(res.status).toBe(400)
      expect(res.body.success).toBe(false)
      expect(res.body.error.details).toContainEqual(
        expect.objectContaining({ field: 'title' })
      )
    })

    it('returns 401 when no auth token is provided', async () => {
      const res = await request(app)
        .post('/api/v1/tasks')
        .send({ title: 'Task', columnId: 'col_123' })

      expect(res.status).toBe(401)
    })
  })
})
```

### Setup helpers (in `setup.ts`)

```typescript
export async function createTestUser(overrides = {}) {
  const password = await bcrypt.hash('testpassword', 10)
  const user = await prisma.user.create({
    data: {
      email: `test-${Date.now()}@example.com`,
      name: 'Test User',
      password,
      ...overrides
    }
  })
  const token = jwt.sign({ id: user.id }, process.env.JWT_SECRET!, { expiresIn: '1h' })
  return { ...user, token }
}

export async function createTestWorkspace(ownerId: string) {
  return prisma.workspace.create({
    data: {
      name: 'Test Workspace',
      slug: `workspace-${Date.now()}`,
      members: { create: { userId: ownerId, role: 'OWNER' } }
    }
  })
}
```

### What to assert in every test

- HTTP status code (exact number)
- `res.body.success` (true or false)
- Shape of `res.body.data` using `toMatchObject` (partial match is fine)
- For list endpoints: `Array.isArray(res.body.data)` and `res.body.meta` presence
- For errors: `res.body.error.message` and `res.body.error.details` when applicable
- Side effects: use `prisma.model.findUnique` to verify DB state after writes

## Frontend test conventions (Vitest + RTL)

### Component test structure

```typescript
import { render, screen, waitFor } from '@testing-library/react'
import userEvent from '@testing-library/user-event'
import { vi } from 'vitest'
import { QueryClient, QueryClientProvider } from '@tanstack/react-query'
import TaskCard from '../output/frontend/src/components/TaskCard'
import type { Task } from '../output/frontend/src/types'

// Wrap component with required providers
function renderWithProviders(ui: React.ReactElement) {
  const queryClient = new QueryClient({ defaultOptions: { queries: { retry: false } } })
  return render(
    <QueryClientProvider client={queryClient}>
      {ui}
    </QueryClientProvider>
  )
}

const mockTask: Task = {
  id: 'task_123',
  title: 'Fix the login bug',
  priority: 'HIGH',
  status: 'IN_PROGRESS',
  createdAt: new Date().toISOString(),
  updatedAt: new Date().toISOString(),
}

describe('TaskCard', () => {
  it('renders the task title', () => {
    renderWithProviders(<TaskCard task={mockTask} />)
    expect(screen.getByText('Fix the login bug')).toBeInTheDocument()
  })

  it('shows the HIGH priority badge', () => {
    renderWithProviders(<TaskCard task={mockTask} />)
    expect(screen.getByText('High')).toBeInTheDocument()
  })

  it('calls onStatusChange when status is updated', async () => {
    const handleStatusChange = vi.fn()
    renderWithProviders(
      <TaskCard task={mockTask} onStatusChange={handleStatusChange} />
    )

    await userEvent.click(screen.getByRole('button', { name: /change status/i }))
    await userEvent.click(screen.getByRole('option', { name: /done/i }))

    expect(handleStatusChange).toHaveBeenCalledWith('task_123', 'DONE')
  })
})
```

### Query conventions

Priority order (use the first one that works):
1. `getByRole` ã most semantic, preferred
2. `getByLabelText` ã for form elements
3. `getByPlaceholderText` ã fallback for inputs
4. `getByText` ã for static text
5. `getByTestId` ã last resort, add `data-testid` attribute

Never use: `container.querySelector`, `getElementsByClassName`, or CSS selectors.

### Mock patterns

```typescript
// Mock an API module
vi.mock('@/api/tasks', () => ({
  getTasks: vi.fn().mockResolvedValue([mockTask]),
  updateTask: vi.fn().mockResolvedValue(mockTask),
}))

// Mock a store
vi.mock('@/stores/authStore', () => ({
  useAuthStore: () => ({
    user: { id: 'user_1', name: 'Test User' },
    token: 'test-token',
    isAuthenticated: true,
  })
}))

// Assert mock was called with correct args
import { updateTask } from '@/api/tasks'
expect(updateTask).toHaveBeenCalledWith('task_123', { status: 'DONE' })
```

## Coverage targets

- Backend: 75% line coverage, 75% function coverage
- Frontend: 70% line coverage, 70% function coverage

These are minimums. Aim higher for critical paths (auth, payment-adjacent, data write).

---

## What NOT to test

- Implementation details (which internal function was called)
- Third-party libraries (don't test that Prisma works)
- Styling (don't assert CSS class names ã assert visible text and roles)
- Private methods (test the public interface only)
- Trivial getters/setters with no logic
